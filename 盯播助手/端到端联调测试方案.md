```python
import requests
import time
import json

# 模拟直播开始
async def simulate_live_start():
    # 发送直播开始消息到RabbitMQ
    import pika
    
    connection = pika.BlockingConnection(
        pika.ConnectionParameters('localhost', 5672, '/', 
                                pika.PlainCredentials('stream_user', 'stream_password')))
    channel = connection.channel()
    
    # 发送直播开始消息
    live_start_message = {
        'type': 'live_status',
        'platform': 'douyu',
        'room_id': '666666',
        'is_live': True,
        'timestamp': '2026-04-09T10:00:00Z'
    }
    
    channel.basic_publish(
        exchange='stream.monitor.exchange',
        routing_key='stream.notification',
        body=json.dumps(live_start_message)
    )
    
    print("直播开始消息已发送")
    connection.close()

if __name__ == "__main__":
    simulate_live_start()
```

```sql
-- 检查任务状态一致性
SELECT 
    t.id,
    t.platform,
    t.room_id,
    t.status as task_status,
    COUNT(p.id) as push_count,
    t.trigger_count,
    t.last_triggered_at
FROM monitoring_tasks t
LEFT JOIN push_logs p ON t.id = p.task_id
WHERE t.platform IN ('bilibili', 'douyu', 'huya')
GROUP BY t.id
ORDER BY t.created_at DESC;

-- 检查消息队列积压
SELECT 
    COUNT(*) as total_barrages,
    platform,
    DATE(created_at) as date
FROM barrage_data
WHERE created_at >= DATE_SUB(NOW(), INTERVAL 1 HOUR)
GROUP BY platform, DATE(created_at);
```

## 🚨 故障排查指南

### 问题1：任务创建成功但爬虫未启动

**可能原因：**
- RabbitMQ连接问题
- Python服务未正确消费消息
- 平台适配器配置错误

**排查步骤：**
```bash
# 1. 检查RabbitMQ连接
docker exec -it stream_monitor_rabbitmq rabbitmqctl list_connections

# 2. 检查Python服务RabbitMQ消费者
docker logs stream_monitor_python | grep -i "rabbitmq\|queue\|consumer"

# 3. 手动发送测试消息
python3 -c "
import pika, json
connection = pika.BlockingConnection(pika.ConnectionParameters('localhost'))
channel = connection.channel()
channel.basic_publish(exchange='stream.monitor.exchange', routing_key='stream.crawler', body=json.dumps({'test': 'message'}))
connection.close()
"
```

### 问题2：AI分析无响应

**可能原因：**
- AI API密钥配置错误
- 网络连接问题
- 请求格式错误

**排查步骤：**
```bash
# 1. 检查AI服务配置
docker exec -it stream_monitor_python cat /app/config/settings.py | grep -i "api"

# 2. 测试AI API连通性
curl -v -X POST "https://api.deepseek.com/chat/completions" \
  -H "Authorization: Bearer $AI_API_KEY" \
  -H "Content-Type: application/json" \
  -d '{"model": "deepseek-chat", "messages": [{"role": "user", "content": "Hello!"}]}' \
  --connect-timeout 10

# 3. 检查Python服务AI模块日志
docker logs stream_monitor_python | grep -i "ai\|openai\|deepseek"
```

### 问题3：通知推送失败

**可能原因：**
- 微信模板配置错误

```bash
# 1. 检查微信配置
```

### 性能测试

```bash
# 压力测试脚本
import asyncio
import aiohttp
import time

async def stress_test():
    tasks = []
    
    # 创建100个监控任务
    for i in range(100):
        task = asyncio.create_task(create_monitoring_task(i))
        tasks.append(task)
        await asyncio.sleep(0.1)  # 控制创建速率
    
    # 等待所有任务完成
    results = await asyncio.gather(*tasks)
    
    print(f"成功创建任务: {sum(results)}/100")

async def create_monitoring_task(task_id):
    try:
        async with aiohttp.ClientSession() as session:
            async with session.post(
                'http://localhost:8080/api/tasks',
                json={
                    'platform': 'bilibili',
                    'roomId': f'room_{task_id}',
                    'taskType': 'keyword_match',
                    'keywords': '["test"]'
                },
                headers={'Authorization': 'Bearer test_token'}
            ) as response:
                return response.status == 200
    except Exception:
        return False

if __name__ == "__main__":
    start_time = time.time()
    asyncio.run(stress_test())
    print(f"压力测试完成，耗时: {time.time() - start_time:.2f}秒")
```
